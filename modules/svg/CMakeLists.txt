
add_env(svg-env
	COMPONENTS
		modules-env
)

	COMPONENTS
		svg-env
		warnings-env
)

target_include_directories(svg-env INTERFACE "${__THIRDPARTY_SVG_PATH}")

add_lib(thirdparty-svg-lib
	PARENT_LIB
		svg-lib
	COMPONENTS
		svg-env	
	SOURCES
		"${__THIRDPARTY_SVG_PATH}/nanosvg.cc"
)

target_glob_sources(svg-lib PRIVATE "*.cpp")
